Choice

Built on select tag with multiple selected


<select class="select" multiple="multiple">
    <option value ="default" selected="selected">default</option>
    <option value ="on">on</option>
    <option value="off">off</option>
</select>
<script type="text/javascript">
    $('.select').asChoice();
</script>

Built on select tag with single selected


<select class="select-radio">
    <option value ="default" selected="selected">default</option>
    <option value ="on" selected="selected">on</option>
    <option value="off">off</option>
</select>
<script type="text/javascript">
    $('.select-radio').asChoice();
</script>

Built on div tag 


<div class="blank"></div>
<script type="text/javascript">
    $('.blank').asChoice({
        status: {
            a: {
                text: 'on',
            },
            b: {
                text: 'off',
            },
            c: {
                text: 'default',
            }
        },
        multiple: false
    });
</script>
